            \begin{array}{l}
                \textbf{proc}\ addword^W(\textbf{value}\ w)\cdot 
                b, W:[ \ \True, b = b_0\And W = W_0\cup \{w\}] \ \\
                \textbf{func}\ checkword^W(\textbf{value}\ w):\mathbb{B}\cdot
                \textbf{var}\ b\cdot\ b, W:[ \ \True, b = (w\in W_0)];\ \textbf{return}\ b\\
                \textbf{proc}\ delword^W(\textbf{value}\ w)\cdot
                b, W:[ w\in W, b = b_0\And W = W_0 \backslash \{w\}]
            \end{array}
        \right)
    )
\end{align*}
